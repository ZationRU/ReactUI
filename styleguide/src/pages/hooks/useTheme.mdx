```tsx
import {useTheme, VStack, Layout, Button, ThemeTokens, HStack} from "@znui/react";

const theme = useTheme();

<VStack
    spacing={5}
    p={20}
    bg={ThemeTokens.surface}
    c={ThemeTokens.primary}
>
    <Layout>
        Current scheme: {theme.currentScheme}
    </Layout>
    <Layout>
        Current resolved scheme: {theme.resolvedScheme}
    </Layout>
    <HStack gap={10}>
        <Button onClick={() => theme.changeScheme('light')}>
            Light scheme
        </Button>
        <Button onClick={() => theme.changeScheme('dark')}>
            Dark scheme
        </Button>
        <Button onClick={() => theme.changeScheme('system')}>
            System scheme
        </Button>
    </HStack>
</VStack>
```
